18  package com.agorapulse.worker.report;
19  
20  import com.agorapulse.worker.JobManager;
21  import com.agorapulse.worker.JobStatus;
22  
23  import java.io.PrintWriter;
24  import java.io.StringWriter;
25  import java.time.Duration;
26  import java.util.Optional;
27  import java.util.Set;
28  
29  public class JobReport {
30  
31      private static final String BORDER = "|%-142s|";
32      private static final String TABLE = "| %-50s | %-8s | %-25s | %-25s | %-20s |";
33  
34      public static String report(JobManager manager) {
35          return report(manager, manager.getJobNames());
36      }
37  
38      public static String report(JobManager manager, Set<String> jobNames) {
39          StringWriter writer = new StringWriter();
40          PrintWriter out = new PrintWriter(writer);
41  
42          out.println(String.format(BORDER, '=').replace(' ', '='));
43          out.println(String.format(TABLE, "Name", "Running", "Last Triggered", "Last Finished", "Took"));
44          out.println(String.format(BORDER, '=').replace(' ', '='));
45          jobNames.stream()
46                  .map(manager::getJob)
47                  .filter(Optional::isPresent)
48                  .map(Optional::get)
49                  .forEach(j -> {
50                      JobStatus status = j.getStatus();
51                      out.println(String.format(
52                              TABLE,
53                              j.getName(),
54                              status.getExecutionCount(),
55                              status.getLastTriggered(),
56                              status.getLastFinished(),
57                              humanReadableFormat(status.getLastDuration())
58                      ));
59                      if (status.getLastException() != null) {
60                          out.println(String.format(BORDER, '-').replace(' ', '-'));
61                          status.getLastException().printStackTrace(out);
62                      }
63                      out.println(String.format(BORDER, '=').replace(' ', '='));
64                  });
65          return writer.toString();
66      }
67  
68      // https://stackoverflow.com/a/40487511/227419
69      public static String humanReadableFormat(Duration duration) {
70          if (duration == null) {
71              return "";
72          }
73          return duration.toString()
74                  .substring(2)
75                  .replaceAll("(\\d[HMS])(?!$)", "$1 ")
76                  .toLowerCase();
77      }
78  
79  }
