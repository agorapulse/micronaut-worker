51  /**
52   * A {@link ExecutableMethodProcessor} for the {@link Job} annotation.
53   * <p>
54   * Based on {@link io.micronaut.scheduling.processor.ScheduledMethodProcessor} with additional option to configure
55   * jobs using configuration properties, reacting on incoming messages from a queue and publishing results of the
56   * execution to a queue.
57   *
58   * @since 1.0
59   */
60  @Singleton
61  public class MethodJobProcessor implements ExecutableMethodProcessor<Job> {
62  
63      private static final Logger LOG = LoggerFactory.getLogger(MethodJobProcessor.class);
64  
65      private static final List<String> SUFFIXES_TO_REMOVE = Arrays.asList("Job", "Consumer", "Producer");
66      private static final String MEMBER_FIXED_RATE = "fixedRate";
67      private static final String MEMBER_INITIAL_DELAY = "initialDelay";
68      private static final String MEMBER_CRON = "cron";
69      private static final String MEMBER_FIXED_DELAY = "fixedDelay";
70      private static final String MEMBER_SCHEDULER = "scheduler";
71  
72      private final BeanContext beanContext;
73      private final TaskExceptionHandler<?, ?> taskExceptionHandler;
74      private final MethodJobInvoker jobMethodInvoker;
75      private final JobManager jobManager;
76      private final ApplicationConfiguration applicationConfiguration;
77      private final JobScheduler jobScheduler;
78      private final ConversionService<?> conversionService;
79      private final WorkerConfiguration workerConfiguration;
80  
81      /**
82       * @param beanContext              The bean context for DI of beans annotated with {@link javax.inject.Inject}
83       * @param taskExceptionHandler     The default task exception handler
84       * @param jobMethodInvoker         The job invoker
85       * @param jobManager               The job manager
86       * @param applicationConfiguration The application configuration
87       * @param jobScheduler             The job scheduler
88       * @param workerConfiguration
89       */
90      @SuppressWarnings("OptionalUsedAsFieldOrParameterType")
91      public MethodJobProcessor(
92          BeanContext beanContext,
93          TaskExceptionHandler<?, ?> taskExceptionHandler,
94          MethodJobInvoker jobMethodInvoker,
95          JobManager jobManager,
96          ApplicationConfiguration applicationConfiguration,
97          JobScheduler jobScheduler,
98          Optional<ConversionService<?>> optionalConversionService,
99          WorkerConfiguration workerConfiguration
100     ) {
101         this.beanContext = beanContext;
102         this.taskExceptionHandler = taskExceptionHandler;
103         this.jobMethodInvoker = jobMethodInvoker;
104         this.jobManager = jobManager;
105         this.applicationConfiguration = applicationConfiguration;
106         this.jobScheduler = jobScheduler;
107         this.conversionService = optionalConversionService.orElse(ConversionService.SHARED);
108         this.workerConfiguration = workerConfiguration;
109     }
110 
111     @Override
112     public void process(BeanDefinition<?> beanDefinition, ExecutableMethod<?, ?> method) {
113         if (!(beanContext instanceof ApplicationContext)) {
114             return;
115         }
116 
117         JobConfiguration configuration = getJobConfiguration(beanDefinition, method);
118 
119         com.agorapulse.worker.Job task = new MethodJob<>(
120             configuration,
121             method,
122             beanDefinition,
123             beanContext,
124             jobMethodInvoker,
125             taskExceptionHandler
126         );
127 
128         jobManager.register(task);
129 
130         if (!configuration.isEnabled()) {
131             LOG.info("Job {} is disabled in the configuration. Remove jobs.{}.enabled = false configuration to re-enable it.", configuration.getName(), configuration.getName());
132             return;
133         }
134 
135         try {
136             jobScheduler.schedule(task);
137         } catch (JobConfigurationException e) {
138             LOG.error("Job declared in method " + method +  " is ignored because it is not correctly configured", e);
139         }
140     }
141 
142     private String getJobName(BeanDefinition<?> beanDefinition, ExecutableMethod<?, ?> method) {
143         Optional<String> valueFromJob = method.stringValue(Job.class).map(NameUtils::hyphenate);
144 
145         if (valueFromJob.isPresent()) {
146             return valueFromJob.get();
147         }
148 
149         Optional<String> valueFromJavaxNamed = method.stringValue("javax.inject.Named").map(NameUtils::hyphenate);
150 
151         if (valueFromJavaxNamed.isPresent()) {
152             return valueFromJavaxNamed.get();
153         }
154 
155         Optional<String> valueFromJakartaNamed = method.stringValue("jakarta.inject.Named").map(NameUtils::hyphenate);
156 
157         if (valueFromJakartaNamed.isPresent()) {
158             return valueFromJakartaNamed.get();
159         }
160 
161         // there are more then one job definition
162         if (beanDefinition.getExecutableMethods().size() > 1) {
163             return JobManager.getDefaultJobName(method.getDeclaringType(), method.getMethodName());
164         }
165 
166         return JobManager.getDefaultJobName(method.getDeclaringType());
167     }
168 
169     private JobConfiguration getJobConfiguration(BeanDefinition<?> beanDefinition, ExecutableMethod<?, ?> method) {
170         String jobName = getJobName(beanDefinition, method);
171 
172         JobConfiguration configurationOverrides = beanContext.findBean(JobConfiguration.class, Qualifiers.byName(jobName))
173             .orElseGet(() -> new DefaultJobConfiguration(jobName, workerConfiguration));
174 
175         DefaultJobConfigurationuration.html#DefaultJobConfiguration">DefaultJobConfiguration configuration = new DefaultJobConfiguration(jobName, workerConfiguration);
176 
177         method.stringValue(Job.class, MEMBER_CRON).ifPresent(configuration::setCron);
178         method.stringValue(Job.class, MEMBER_FIXED_DELAY).ifPresent(fixedDelay -> configuration.setFixedDelay(convertDuration(jobName, fixedDelay, "fixed delay")));
179         method.stringValue(Job.class, MEMBER_FIXED_RATE).ifPresent(fixedRate -> configuration.setFixedRate(convertDuration(jobName, fixedRate, "fixed rate")));
180         method.stringValue(Job.class, MEMBER_INITIAL_DELAY).ifPresent(initialDelay -> configuration.setInitialDelay(convertDuration(jobName, initialDelay, "initial delay")));
181         method.stringValue(Job.class, MEMBER_SCHEDULER).ifPresent(configuration::setScheduler);
182 
183 
184         configuration.setLeaderOnly(method.findAnnotation(LeaderOnly.class).isPresent());
185         configuration.setFollowerOnly(method.findAnnotation(FollowerOnly.class).isPresent());
186 
187         method.findAnnotation(Concurrency.class).flatMap(a -> a.getValue(Integer.class)).ifPresent(configuration::setConcurrency);
188         method.findAnnotation(Fork.class).flatMap(a -> a.getValue(Integer.class)).ifPresent(configuration::setFork);
189 
190         configureConsumerQueue(jobName, method.findAnnotation(Consumes.class), configuration.getConsumer());
191         configureQueue(method.findAnnotation(Produces.class), configuration.getProducer());
192 
193         configuration.mergeWith(configurationOverrides);
194 
195         if (configuration.getConsumer().getQueueName() == null) {
196             configuration.getConsumer().setQueueName(extractQueueNameFromMethod(method));
197         }
198 
199         if (configuration.getProducer().getQueueName() == null) {
200             configuration.getProducer().setQueueName(extractQueueNameFromMethod(method));
201         }
202 
203         return configuration;
204     }
205 
206     private <A extends Annotation> void configureQueue(Optional<AnnotationValue<A>> consumesAnnotation, MutableJobConfiguration.MutableQueueConfiguration queueConfiguration) {
207         if (consumesAnnotation.isPresent()) {
208             AnnotationValue<A> annotationValue = consumesAnnotation.get();
209             queueConfiguration.setQueueName(annotationValue.getRequiredValue(String.class));
210 
211             annotationValue.stringValue("type").ifPresent(type -> {
212                 if (StringUtils.isNotEmpty(type)) {
213                     queueConfiguration.setQueueType(type);
214                 }
215             });
216         }
217     }
218 
219     private void configureConsumerQueue(String jobName, Optional<AnnotationValue<Consumes>> consumesAnnotation, MutableJobConfiguration.MutableConsumerQueueConfiguration queueConfiguration) {
220         if (consumesAnnotation.isPresent()) {
221             configureQueue(consumesAnnotation, queueConfiguration);
222 
223             AnnotationValue<Consumes> annotationValue = consumesAnnotation.get();
224 
225             annotationValue.stringValue("waitingTime").ifPresent(waitingTime -> {
226                 if (StringUtils.isNotEmpty(waitingTime)) {
227                     queueConfiguration.setWaitingTime(convertDuration(jobName, waitingTime, "waiting time"));
228                 }
229             });
230 
231             annotationValue.intValue("maxMessages").ifPresent(maxMessages -> {
232                 if (maxMessages > 1) {
233                     queueConfiguration.setMaxMessages(maxMessages);
234                 }
235             });
236         }
237     }
238 
239     private String extractQueueNameFromMethod(ExecutableMethod<?, ?> method) {
240         String name = SUFFIXES_TO_REMOVE
241             .stream()
242             .reduce(
243                 method.getDeclaringType().getSimpleName(),
244                 (acc, suffix) -> acc.endsWith(suffix) ? acc.substring(0, acc.length() - suffix.length()) : acc
245             );
246 
247         String queueName = applicationConfiguration.getName().map(n -> n + "_" + name).orElse(name);
248         if (LOG.isTraceEnabled()) {
249             LOG.trace("Using default queue name {} queue for method {}#{}. Use @Consumes or @Produces to specify different name!", queueName, method.getDeclaringType().getName(), method.getMethodName());
250         }
251         return queueName;
252     }
253 
254     private Duration convertDuration(String jobName, String durationString, String humanReadableProperty) {
255         Optional<Duration> converted = conversionService.convert(durationString, Duration.class);
256 
257         if (converted.isPresent()) {
258             return converted.get();
259         }
260 
261         throw new IllegalArgumentException("Invalid " + humanReadableProperty + "  definition for job " + jobName + " : " + durationString);
262     }
