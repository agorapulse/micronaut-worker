18  package com.agorapulse.worker.schedule;
19  
20  import com.agorapulse.worker.JobConfiguration;
21  import com.agorapulse.worker.JobConfigurationException;
22  import com.agorapulse.worker.JobScheduler;
23  import io.micronaut.context.BeanContext;
24  import io.micronaut.context.annotation.Requires;
25  import io.micronaut.core.util.StringUtils;
26  import io.micronaut.inject.qualifiers.Qualifiers;
27  import io.micronaut.scheduling.ScheduledExecutorTaskScheduler;
28  import io.micronaut.scheduling.TaskScheduler;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  
32  import javax.inject.Singleton;
33  import java.io.Closeable;
34  import java.time.Duration;
35  import java.util.Optional;
36  import java.util.Queue;
37  import java.util.concurrent.ConcurrentLinkedDeque;
38  import java.util.concurrent.ExecutorService;
39  import java.util.concurrent.ScheduledExecutorService;
40  import java.util.concurrent.ScheduledFuture;
41  
42  @Singleton
43  @Requires(property = "jobs.enabled", notEquals = "false")
44  public class DefaultJobScheduler implements JobScheduler, Closeable {
45  
46      private static final Logger LOG = LoggerFactory.getLogger(DefaultJobScheduler.class);
47  
48      private final BeanContext beanContext;
49      private final Queue<ScheduledFuture<?>> scheduledTasks = new ConcurrentLinkedDeque<>();
50  
51      /**
52       * @param beanContext               The bean context for DI of beans annotated with {@link javax.inject.Inject}
53       */
54      public DefaultJobScheduler(
55              BeanContext beanContext
56      ) {
57          this.beanContext = beanContext;
58      }
59  
60      @Override
61      public void close() {
62          for (ScheduledFuture<?> scheduledTask : scheduledTasks) {
63              if (!scheduledTask.isCancelled()) {
64                  scheduledTask.cancel(false);
65              }
66          }
67      }
68  
69      @Override
70      public void schedule(com.agorapulse.worker.Job job) {
71          JobConfiguration configuration = job.getConfiguration();
72          TaskScheduler taskScheduler = getTaskScheduler(job);
73  
74          Duration initialDelay = configuration.getInitialDelay();
75  
76          if (StringUtils.isNotEmpty(configuration.getCron())) {
77              if (LOG.isWarnEnabled() && initialDelay != null && !initialDelay.isZero()) {
78                  LOG.warn("Ignoring initial delay {} of cron job {} [{}] for {}", initialDelay, configuration.getName(), configuration.getCron(), job.getSource());
79              }
80              if (LOG.isDebugEnabled()) {
81                  LOG.debug("Scheduling cron job {} [{}] for {}", configuration.getName(), configuration.getCron(), job.getSource());
82              }
83              taskScheduler.schedule(configuration.getCron(), job);
84          } else if (configuration.getFixedRate() != null) {
85              Duration duration = configuration.getFixedRate();
86              if (LOG.isDebugEnabled()) {
87                  LOG.debug("Scheduling fixed rate job {} [{}] for {}", configuration.getName(), duration, job.getSource());
88              }
89  
90              ScheduledFuture<?> scheduledFuture = taskScheduler.scheduleAtFixedRate(initialDelay, duration, job);
91              scheduledTasks.add(scheduledFuture);
92          } else if (configuration.getFixedDelay() != null) {
93              Duration duration = configuration.getFixedDelay();
94  
95              if (LOG.isDebugEnabled()) {
96                  LOG.debug("Scheduling fixed delay task {} [{}] for {}", configuration.getName(), duration, job.getSource());
97              }
98  
99              ScheduledFuture<?> scheduledFuture = taskScheduler.scheduleWithFixedDelay(initialDelay, duration, job);
100             scheduledTasks.add(scheduledFuture);
101         } else if (initialDelay != null) {
102             ScheduledFuture<?> scheduledFuture = taskScheduler.schedule(initialDelay, job);
103 
104             scheduledTasks.add(scheduledFuture);
105         } else {
106             throw new JobConfigurationException(job, "Failed to schedule job " + configuration.getName() + " declared in "  + job.getSource() + ". Invalid definition");
107         }
108     }
109 
110     private TaskScheduler getTaskScheduler(com.agorapulse.worker.Job job) {
111         JobConfiguration configuration = job.getConfiguration();
112         Optional<TaskScheduler> optionalTaskScheduler = beanContext.findBean(TaskScheduler.class, Qualifiers.byName(configuration.getScheduler()));
113 
114         if (!optionalTaskScheduler.isPresent()) {
115             optionalTaskScheduler = beanContext.findBean(ExecutorService.class, Qualifiers.byName(configuration.getScheduler()))
116                     .filter(ScheduledExecutorService.class::isInstance)
117                     .map(ScheduledExecutorTaskScheduler::new);
118         }
119 
120         return optionalTaskScheduler.orElseThrow(() -> new JobConfigurationException(job, "No scheduler of type TaskScheduler configured for name: " + configuration.getScheduler()));
121     }
122 }
