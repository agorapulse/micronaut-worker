18 package com.agorapulse.worker.tck.executor
19 
20 import com.agorapulse.worker.executor.DistributedJobExecutor
21 import io.micronaut.context.ApplicationContext
22 import spock.lang.Retry
23 import spock.lang.Specification
24 
25 abstract class AbstractJobExecutorSpec extends Specification {
26 
27     public static final String CONCURRENT_JOB_TEST_ENVIRONMENT = 'concurrent-job-test'
28 
29     public static final String JOBS_INITIAL_DELAY = '100ms'
30     public static final long LONG_RUNNING_JOB_DURATION = 500
31     public static final long SLEEP_BEFORE_CHECKING = 2000
32 
33     @Retry(count = 10)
34     void 'jobs executed appropriate times on three servers'() {
35         given:
36             ApplicationContext one = buildContext()
37             ApplicationContext two = buildContext()
38             ApplicationContext three = buildContext()
39 
40         expect:
41             requiredExecutorType.isInstance(one.getBean(DistributedJobExecutor))
42 
43         when:
44             LongRunningJob jobOne = one.getBean(LongRunningJob)
45             LongRunningJob jobTwo = two.getBean(LongRunningJob)
46             LongRunningJob jobThree = three.getBean(LongRunningJob)
47 
48             Thread.sleep(SLEEP_BEFORE_CHECKING)
49 
50             List<LongRunningJob> jobs = [jobOne, jobTwo, jobThree]
51         then:
52             // jobs are unique
53             jobs.unique().size() == 3
54 
55             // unlimited jobs are executed on every server
56             jobs.count { it.unlimited.get() == 1 } == 3
57 
58             // concurrent jobs are at most n-times
59             jobs.count { it.concurrent.get() == 1 } == 2
60 
61             // leader job is executed only on leader
62             jobs.count { it.leader.get() == 1 } == 1
63 
64             // follower job is executed only on followers
65             jobs.count { it.follower.get() == 1  } == expectedFollowersCount
66 
67             //  consecutive job is only executed once on a random server
68             jobs.count { it.consecutive.get() == 1 } == 1
69 
70             // producer job is executed only on leader
71             jobs.count { it.producer.get() == 1 } == 1
72         cleanup:
73             closeQuietly one, two, three
74     }
75 
76     protected abstract ApplicationContext buildContext()
77     protected abstract Class<?> getRequiredExecutorType()
78 
79     // some implementation may not support followers, such as the local implementation
80     protected int getExpectedFollowersCount() {
81         return 2
82     }
83 
84     @SuppressWarnings([
85         'CatchException',
86     ])
87     private static void closeQuietly(ApplicationContext... contexts) {
88         for (ApplicationContext c : contexts) {
89             try {
90                 if (c.running) {
91                     c.close()
92                 }
93             } catch (Exception ignored) {
94                 // ignore
95             }
96         }
97     }
98 
99 }
