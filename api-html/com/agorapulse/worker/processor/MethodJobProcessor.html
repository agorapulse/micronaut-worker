049 /**
050  * A {@link ExecutableMethodProcessor} for the {@link Job} annotation.
051  * <p>
052  * Based on {@link io.micronaut.scheduling.processor.ScheduledMethodProcessor} with additional option to configure
053  * jobs using configuration properties, reacting on incoming messages from a queue and publishing results of the
054  * execution to a queue.
055  *
056  * @since 1.0
057  */
058 @Singleton
059 public class MethodJobProcessor implements ExecutableMethodProcessor<Job> {
060 
061     private static final Logger LOG = LoggerFactory.getLogger(MethodJobProcessor.class);
062 
063     private static final List<String> SUFFIXES_TO_REMOVE = Arrays.asList("Job", "Consumer", "Producer");
064     private static final String MEMBER_FIXED_RATE = "fixedRate";
065     private static final String MEMBER_INITIAL_DELAY = "initialDelay";
066     private static final String MEMBER_CRON = "cron";
067     private static final String MEMBER_FIXED_DELAY = "fixedDelay";
068     private static final String MEMBER_SCHEDULER = "scheduler";
069 
070     private final BeanContext beanContext;
071     private final TaskExceptionHandler<?, ?> taskExceptionHandler;
072     private final MethodJobInvoker jobMethodInvoker;
073     private final JobManager jobManager;
074     private final ApplicationConfiguration applicationConfiguration;
075     private final JobScheduler jobScheduler;
076     private final ConversionService<?> conversionService;
077     private final WorkerConfiguration workerConfiguration;
078 
079     /**
080      * @param beanContext              The bean context for DI of beans annotated with {@link javax.inject.Inject}
081      * @param taskExceptionHandler     The default task exception handler
082      * @param jobMethodInvoker         The job invoker
083      * @param jobManager               The job manager
084      * @param applicationConfiguration The application configuration
085      * @param jobScheduler             The job scheduler
086      * @param workerConfiguration
087      */
088     @SuppressWarnings("OptionalUsedAsFieldOrParameterType")
089     public MethodJobProcessor(
090         BeanContext beanContext,
091         TaskExceptionHandler<?, ?> taskExceptionHandler,
092         MethodJobInvoker jobMethodInvoker,
093         JobManager jobManager,
094         ApplicationConfiguration applicationConfiguration,
095         JobScheduler jobScheduler,
096         Optional<ConversionService<?>> optionalConversionService,
097         WorkerConfiguration workerConfiguration
098     ) {
099         this.beanContext = beanContext;
100         this.taskExceptionHandler = taskExceptionHandler;
101         this.jobMethodInvoker = jobMethodInvoker;
102         this.jobManager = jobManager;
103         this.applicationConfiguration = applicationConfiguration;
104         this.jobScheduler = jobScheduler;
105         this.conversionService = optionalConversionService.orElse(ConversionService.SHARED);
106         this.workerConfiguration = workerConfiguration;
107     }
108 
109     @Override
110     public void process(BeanDefinition<?> beanDefinition, ExecutableMethod<?, ?> method) {
111         if (!(beanContext instanceof ApplicationContext)) {
112             return;
113         }
114 
115         JobConfiguration configuration = getJobConfiguration(beanDefinition, method);
116 
117         com.agorapulse.worker.Job task = new MethodJob<>(
118             configuration,
119             method,
120             beanDefinition,
121             beanContext,
122             jobMethodInvoker,
123             taskExceptionHandler
124         );
125 
126         jobManager.register(task);
127 
128         if (!configuration.isEnabled()) {
129             LOG.info("Job {} is disabled in the configuration. Remove jobs.{}.enabled = false configuration to re-enable it.", configuration.getName(), configuration.getName());
130             return;
131         }
132 
133         jobScheduler.schedule(task);
134     }

136     private String getJobName(BeanDefinition<?> beanDefinition, ExecutableMethod<?, ?> method) {
137         Optional<String> valueFromJob = method.stringValue(Job.class).map(NameUtils::hyphenate);
138 
139         if (valueFromJob.isPresent()) {
140             return valueFromJob.get();
141         }
142 
143         Optional<String> valueFromJavaxNamed = method.stringValue("javax.inject.Named").map(NameUtils::hyphenate);
144 
145         if (valueFromJavaxNamed.isPresent()) {
146             return valueFromJavaxNamed.get();
147         }
148 
149         Optional<String> valueFromJakartaNamed = method.stringValue("jakarta.inject.Named").map(NameUtils::hyphenate);
150 
151         if (valueFromJakartaNamed.isPresent()) {
152             return valueFromJakartaNamed.get();
153         }
154 
155         // there are more then one job definition
156         if (beanDefinition.getExecutableMethods().size() > 1) {
157             return NameUtils.hyphenate(method.getDeclaringType().getSimpleName() + "-" + method.getMethodName());
158         }
159 
160         return NameUtils.hyphenate(method.getDeclaringType().getSimpleName());
161     }
162 
163     private JobConfiguration getJobConfiguration(BeanDefinition<?> beanDefinition, ExecutableMethod<?, ?> method) {
164         String jobName = getJobName(beanDefinition, method);
165 
166         JobConfiguration configurationOverrides = beanContext.findBean(JobConfiguration.class, Qualifiers.byName(jobName))
167             .orElseGet(() -> new DefaultJobConfiguration(jobName, workerConfiguration));
168 
169         DefaultJobConfiguration configuration = new DefaultJobConfiguration(jobName, workerConfiguration);
170 
171         method.stringValue(Job.class, MEMBER_CRON).ifPresent(configuration::setCron);
172         method.stringValue(Job.class, MEMBER_FIXED_DELAY).ifPresent(fixedDelay -> configuration.setFixedDelay(convertDuration(jobName, fixedDelay, "fixed delay")));
173         method.stringValue(Job.class, MEMBER_FIXED_RATE).ifPresent(fixedRate -> configuration.setFixedRate(convertDuration(jobName, fixedRate, "fixed rate")));
174         method.stringValue(Job.class, MEMBER_INITIAL_DELAY).ifPresent(initialDelay -> configuration.setInitialDelay(convertDuration(jobName, initialDelay, "initial delay")));
175         method.stringValue(Job.class, MEMBER_SCHEDULER).ifPresent(configuration::setScheduler);
176 
177 
178         configuration.setLeaderOnly(method.findAnnotation(LeaderOnly.class).isPresent());
179         configuration.setFollowerOnly(method.findAnnotation(FollowerOnly.class).isPresent());
180 
181         method.findAnnotation(Concurrency.class).flatMap(a -> a.getValue(Integer.class)).ifPresent(configuration::setConcurrency);
182 
183         configureConsumerQueue(jobName, method.findAnnotation(Consumes.class), configuration.getConsumer());
184         configureQueue(method.findAnnotation(Produces.class), configuration.getProducer());
185 
186         configuration.mergeWith(configurationOverrides);
187 
188         if (configuration.getConsumer().getQueueName() == null) {
189             configuration.getConsumer().setQueueName(extractQueueNameFromMethod(method));
190         }
191 
192         if (configuration.getProducer().getQueueName() == null) {
193             configuration.getProducer().setQueueName(extractQueueNameFromMethod(method));
194         }
195 
196         return configuration;
197     }
198 
199     private <A extends Annotation> void configureQueue(Optional<AnnotationValue<A>> consumesAnnotation, MutableJobConfiguration.MutableQueueConfiguration queueConfiguration) {
200         if (consumesAnnotation.isPresent()) {
201             AnnotationValue<A> annotationValue = consumesAnnotation.get();
202             queueConfiguration.setQueueName(annotationValue.getRequiredValue(String.class));
203 
204             annotationValue.stringValue("type").ifPresent(type -> {
205                 if (StringUtils.isNotEmpty(type)) {
206                     queueConfiguration.setQueueType(type);
207                 }
208             });
209         }
210     }
211 
212     private void configureConsumerQueue(String jobName, Optional<AnnotationValue<Consumes>> consumesAnnotation, MutableJobConfiguration.MutableConsumerQueueConfiguration queueConfiguration) {
213         if (consumesAnnotation.isPresent()) {
214             configureQueue(consumesAnnotation, queueConfiguration);
215 
216             AnnotationValue<Consumes> annotationValue = consumesAnnotation.get();
217 
218             annotationValue.stringValue("waitingTime").ifPresent(waitingTime -> {
219                 if (StringUtils.isNotEmpty(waitingTime)) {
220                     queueConfiguration.setWaitingTime(convertDuration(jobName, waitingTime, "waiting time"));
221                 }
222             });
223 
224             annotationValue.intValue("maxMessages").ifPresent(maxMessages -> {
225                 if (maxMessages > 1) {
226                     queueConfiguration.setMaxMessages(maxMessages);
227                 }
228             });
229         }
230     }
231 
232     private String extractQueueNameFromMethod(ExecutableMethod<?, ?> method) {
233         String name = SUFFIXES_TO_REMOVE
234             .stream()
235             .reduce(
236                 method.getDeclaringType().getSimpleName(),
237                 (acc, suffix) -> acc.endsWith(suffix) ? acc.substring(0, acc.length() - suffix.length()) : acc
238             );
239 
240         String queueName = applicationConfiguration.getName().map(n -> n + "_" + name).orElse(name);
241         if (LOG.isTraceEnabled()) {
242             LOG.trace("Using default queue name {} queue for method {}#{}. Use @Consumes or @Produces to specify different name!", queueName, method.getDeclaringType().getName(), method.getMethodName());
243         }
244         return queueName;
245     }
246 
247     private Duration convertDuration(String jobName, String durationString, String humanReadableProperty) {
248         Optional<Duration> converted = conversionService.convert(durationString, Duration.class);
249 
250         if (converted.isPresent()) {
251             return converted.get();
252         }
253 
254         throw new IllegalArgumentException("Invalid " + humanReadableProperty + "  definition for job " + jobName + " : " + durationString);
255     }
