018 package com.agorapulse.worker.processor;
019 
020 import com.agorapulse.worker.Job;
021 import com.agorapulse.worker.JobConfiguration;
022 import com.agorapulse.worker.JobConfigurationException;
023 import com.agorapulse.worker.event.JobExecutionFinishedEvent;
024 import com.agorapulse.worker.event.JobExecutionResultEvent;
025 import com.agorapulse.worker.event.JobExecutionStartedEvent;
026 import com.agorapulse.worker.executor.DistributedJobExecutor;
027 import com.agorapulse.worker.queue.JobQueues;
028 import io.micronaut.context.BeanContext;
029 import io.micronaut.context.event.ApplicationEventPublisher;
030 import io.micronaut.inject.ExecutableMethod;
031 import io.micronaut.inject.qualifiers.Qualifiers;
032 import io.reactivex.Flowable;
033 import io.reactivex.Maybe;
034 import org.reactivestreams.Publisher;
035 import org.slf4j.Logger;
036 import org.slf4j.LoggerFactory;
037 
038 import jakarta.inject.Singleton;
039 import java.time.Duration;
040 import java.util.Optional;
041 import java.util.concurrent.Callable;
042 import java.util.function.Function;
043 
044 @Singleton
045 public class DefaultMethodJobInvoker implements MethodJobInvoker {
046 
047     private static final Logger LOGGER = LoggerFactory.getLogger(Job.class);
048 
049     private final BeanContext context;
050     private final ApplicationEventPublisher applicationEventPublisher;
051     private final DistributedJobExecutor distributedJobExecutor;
052 
053     public DefaultMethodJobInvoker(
054         BeanContext context,
055         ApplicationEventPublisher applicationEventPublisher,
056         DistributedJobExecutor distributedJobExecutor
057     ) {
058         this.context = context;
059         this.applicationEventPublisher = applicationEventPublisher;
060         this.distributedJobExecutor = distributedJobExecutor;
061     }
062 
063     @SuppressWarnings("unchecked")
064     public <B> void invoke(MethodJob<B, ?> job, B bean) {
065         ExecutableMethod<B, ?> method = job.getMethod();
066         JobConfiguration configuration = job.getConfiguration();
067 
068         if (method.getArguments().length > 1) {
069             throw new JobConfigurationException(job, "Cannot have more than one argument in a method annotated with @Job");
070         }
071 
072         boolean consumer = method.getArguments().length == 1;
073         boolean producer = !method.getReturnType().getType().equals(void.class);
074 
075         boolean leaderOnly = producer && !consumer || configuration.isLeaderOnly();
076         boolean followerOnly = configuration.isFollowerOnly();
077         int concurrency = configuration.getConcurrency();
078 
079         if (leaderOnly && followerOnly) {
080             throw new JobConfigurationException(job, "Cannot use @FollowerOnly on a producer method or method annotated with @LeaderOnly");
081         }
082 
083         Function<Callable<Object>, Publisher<Object>> executor = executor(configuration.getName(), leaderOnly, followerOnly, concurrency);
084 
085         applicationEventPublisher.publishEvent(new JobExecutionStartedEvent(
086             configuration.getName()
087         ));
088 
089         if (method.getArguments().length == 0) {
090             handleResult(configuration, executor.apply(() -> method.invoke(bean)));
091         } else if (method.getArguments().length == 1) {
092             JobConfiguration.ConsumerQueueConfiguration queueConfiguration = configuration.getConsumer();
093             queues(queueConfiguration.getQueueType()).readMessages(
094                 queueConfiguration.getQueueName(),
095                 queueConfiguration.getMaxMessages() < 1 ? 1 : queueConfiguration.getMaxMessages(),
096                 Optional.ofNullable(queueConfiguration.getWaitingTime()).orElse(Duration.ZERO),
097                 method.getArguments()[0],
098                 message -> handleResult(configuration, executor.apply(() -> method.invoke(bean, message)))
099             );
100         } else {
101             LOGGER.error("Too many arguments for " + method + "! The job method wasn't executed!");
102         }
103 
104         applicationEventPublisher.publishEvent(new JobExecutionFinishedEvent(
105             configuration.getName()
106         ));
107     }
108 
109     private <T> Function<Callable<T>, Publisher<T>> executor(String jobName, boolean leaderOnly, boolean followerOnly, int concurrency) {
110         if (concurrency > 0) {
111             return s -> distributedJobExecutor.executeConcurrently(jobName, concurrency, s);
112         }
113         if (leaderOnly) {
114             return s -> distributedJobExecutor.executeOnlyOnLeader(jobName, s);
115         }
116         if (followerOnly) {
117             return s -> distributedJobExecutor.executeOnlyOnFollower(jobName, s);
118         }
119         return s -> Maybe.fromCallable(s).toFlowable();
120     }
121 
122     protected void handleResult(JobConfiguration configuration, Publisher<Object> resultPublisher) {
123         Object result = Flowable.fromPublisher(resultPublisher).blockingFirst(null);
124 
125         applicationEventPublisher.publishEvent(new JobExecutionResultEvent(
126             configuration.getName(),
127             result
128         ));
129 
130         if (result == null) {
131             return;
132         }
133 
134         String queueName = configuration.getProducer().getQueueName();
135 
136         JobQueues sender = queues(configuration.getProducer().getQueueType());
137 
138         if (result instanceof Publisher) {
139             Flowable<?> publisher = Flowable.fromPublisher((Publisher<?>) result);
140             publisher.subscribe(o -> {
141                 if (LOGGER.isDebugEnabled()) {
142                     LOGGER.debug("Sending message {} to {} using {}", o, queueName, sender);
143 
144                 }
145                 sender.sendMessage(queueName, o);
146             }, t -> LOGGER.error("Exception sending messages to queue " + queueName, t));
147             return;
148         }
149 
150         if (LOGGER.isDebugEnabled()) {
151             LOGGER.debug("Sending message {} to {} using {}", result, queueName, sender);
152 
153         }
154 
155         sender.sendMessage(queueName, result);
156     }
157 
158     private JobQueues queues(String qualifier) {
159         return context.findBean(
160             JobQueues.class,
161             qualifier == null ? null : Qualifiers.byName(qualifier)
162         )
163             .orElseGet(() -> context.getBean(JobQueues.class));
164     }
