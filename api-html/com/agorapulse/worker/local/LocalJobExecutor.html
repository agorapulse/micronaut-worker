18 package com.agorapulse.worker.local;
19 
20 import com.agorapulse.worker.executor.DistributedJobExecutor;
21 import io.micronaut.context.annotation.Secondary;
22 import io.reactivex.Maybe;
23 import org.reactivestreams.Publisher;
24 import org.slf4j.Logger;
25 import org.slf4j.LoggerFactory;
26 
27 import jakarta.inject.Named;
28 import jakarta.inject.Singleton;
29 import java.util.concurrent.Callable;
30 import java.util.concurrent.ConcurrentHashMap;
31 import java.util.concurrent.ConcurrentMap;
32 import java.util.concurrent.ExecutorService;
33 import java.util.concurrent.atomic.AtomicInteger;
34 
35 @Secondary
36 @Singleton
37 public class LocalJobExecutor implements DistributedJobExecutor {
38 
39     private static final Logger LOGGER = LoggerFactory.getLogger(LocalJobExecutor.class);
40 
41     private final ConcurrentMap<String, AtomicInteger> counts = new ConcurrentHashMap<>();
42     private final ExecutorService executorService;
43 
44     public LocalJobExecutor(@Named("local-job-executor") ExecutorService executorService) {
45         this.executorService = executorService;
46     }
47 
48     @Override
49     public <R> Publisher<R> executeOnlyOnLeader(String jobName, Callable<R> supplier) {
50         return executeConcurrently(jobName, 1, supplier);
51     }
52 
53     @Override
54     public <R> Publisher<R> executeConcurrently(String jobName, int concurrency, Callable<R> supplier) {
55         return Maybe.fromFuture(executorService.submit(() -> {
56             int increasedCount = counts.computeIfAbsent(jobName, s -> new AtomicInteger(0)).incrementAndGet();
57             LOGGER.trace("Increased count for job {} limited to {}: {}", jobName, concurrency, increasedCount);
58             if (increasedCount > concurrency) {
59                 counts.get(jobName).decrementAndGet();
60                 return null;
61             }
62 
63             R result = supplier.call();
64             int decreasedCount = counts.get(jobName).decrementAndGet();
65             LOGGER.trace("Decreased count for job {} limited to {}: {}", jobName, concurrency, decreasedCount);
66             return result;
67         })).toFlowable();
68     }
69 
70     @Override
71     public <R> Publisher<R> executeOnlyOnFollower(String jobName, Callable<R> supplier) {
72         return Maybe.fromFuture(executorService.submit(supplier)).toFlowable();
73     }
74 
75 }
