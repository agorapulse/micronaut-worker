018 package com.agorapulse.worker.queues.redis;
019 
020 import com.agorapulse.worker.queue.JobQueues;
021 import com.fasterxml.jackson.core.JsonProcessingException;
022 import com.fasterxml.jackson.databind.ObjectMapper;
023 import io.lettuce.core.RedisClient;
024 import io.lettuce.core.TransactionResult;
025 import io.lettuce.core.api.StatefulRedisConnection;
026 import io.lettuce.core.api.async.RedisAsyncCommands;
027 import io.lettuce.core.api.sync.RedisCommands;
028 import io.lettuce.core.support.BoundedAsyncPool;
029 import io.lettuce.core.support.BoundedPoolConfig;
030 import io.micronaut.core.type.Argument;
031 import io.micronaut.jackson.JacksonConfiguration;
032 import org.slf4j.Logger;
033 import org.slf4j.LoggerFactory;
034 
035 import java.time.Duration;
036 import java.util.List;
037 import java.util.concurrent.ExecutionException;
038 import java.util.function.Consumer;
039 
040 public class RedisQueues implements JobQueues {
041 
042     private static final Logger LOGGER = LoggerFactory.getLogger(RedisQueues.class);
043     private static final String PREFIX_DATED_QUEUE = "DATED_QUEUE::";
044 
045     private final ObjectMapper objectMapper;
046     private final BoundedAsyncPool<StatefulRedisConnection<String, String>> pool;
047 
048     public RedisQueues(ObjectMapper objectMapper, RedisClient client, RedisPoolConfiguration redisPoolConfiguration) {
049         this.objectMapper = objectMapper;
050 
051         BoundedPoolConfig config = BoundedPoolConfig
052             .builder()
053             .minIdle(redisPoolConfiguration.getMinIdle())
054             .maxIdle(redisPoolConfiguration.getMaxIdle())
055             .maxTotal(redisPoolConfiguration.getMaxTotal())
056             .testOnAcquire(redisPoolConfiguration.isTestOnAcquire())
057             .testOnCreate(redisPoolConfiguration.isTestOnCreate())
058             .testOnRelease(redisPoolConfiguration.isTestOnRelease())
059             .build();
060 
061         pool = new BoundedAsyncPool<>(new ConnectionFactory(client), config);
062     }
063 
064     @Override
065     public <T> void readMessages(String queueName, int maxNumberOfMessages, Duration waitTime, Argument<T> argument, Consumer<T> action) {
066         TransactionResult result = withTransaction(redisCommands -> {
067             String key = getKey(queueName);
068             redisCommands.zrange(key, 0, maxNumberOfMessages - 1);
069             redisCommands.zremrangebyrank(key, 0, maxNumberOfMessages - 1);
070         });
071 
072         if (result == null) {
073             return;
074         }
075 
076         List<String> messages = result.get(0);
077 
078         messages.forEach(body -> {
079             try {
080                 action.accept(objectMapper.readValue(body, JacksonConfiguration.constructType(argument, objectMapper.getTypeFactory())));
081             } catch (JsonProcessingException e) {
082                 throw new IllegalArgumentException("Cannot convert to " + argument + "from message\n" + body, e);
083             }
084         });
085     }
086 
087     @Override
088     public void sendMessage(String queueName, Object result) {
089         try {
090             String item = objectMapper.writeValueAsString(result);
091             withRedis(redisCommands -> {
092                 String key = getKey(queueName);
093                 redisCommands.zscore(key, item).thenAccept(zscore -> {
094                     if (zscore == null) {
095                         long time = System.currentTimeMillis();
096                         redisCommands.zadd(key, time, item);
097                     }
098                 });
099             });
100         } catch (JsonProcessingException e) {
101             throw new IllegalArgumentException("Cannot write " + result + " to JSON", e);
102         }
103     }
104 
105     private String getKey(String queueName) {
106         return PREFIX_DATED_QUEUE + queueName;
107     }
108 
109     private TransactionResult withTransaction(Consumer<RedisCommands<String, String>> action) {
110         try {
111             StatefulRedisConnection<String, String> connection = pool.acquire().get();
112             RedisCommands<String, String> sync = connection.sync();
113             try {
114                 sync.multi();
115                 action.accept(sync);
116                 return sync.exec();
117             } finally {
118                 pool.release(connection);
119             }
120         } catch (InterruptedException | ExecutionException e) {
121             LOGGER.error("Exception obtaining connection from the pool", e);
122             return null;
123         }
124     }
125 
126     private void withRedis(
127         Consumer<RedisAsyncCommands<String, String>> action
128     ) {
129         try {
130             StatefulRedisConnection<String, String> connection = pool.acquire().get();
131             RedisAsyncCommands<String, String> sync = connection.async();
132             try {
133                 action.accept(sync);
134             } finally {
135                 pool.release(connection);
136             }
137         } catch (InterruptedException | ExecutionException e) {
138             LOGGER.error("Exception obtaining connection from the pool", e);
139         }
140     }
